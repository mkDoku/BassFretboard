\documentclass[crop,tikz]{standalone}% 'crop' is the default for v1.0, before it was 'preview'



% musical symbols
% \fl -> flat
% \sh -> sharp
\usepackage{musicography}

% tikz
\usepackage{tikz}
\usetikzlibrary{
    positioning
  , shapes
  , calc
}

\usepackage{lmodern}
\usepackage[T1]{fontenc}


\begin{document}
\tikzset{
  notePos/.style={
      draw
    , circle
    , thick
    , minimum size=1.5cm
  }
}

\begin{tikzpicture}[]

  % strings
  \node[notePos] (GString) {G};
  \node[notePos, below=of GString] (DString) {D};
  \node[notePos, below=of DString] (AString) {A};
  \node[notePos, below=of AString] (EString) {E};

  % 1st fret
  \node[notePos, right=of GString] (1Aflat) {A\fl / G\sh};
  \node[notePos, right=of DString] (1Eflat) {E};
  \node[notePos, right=of AString] (1Bflat) {B};
  \node[notePos, right=of EString] (1Fflat) {F};
  
  % fret board beginning
  \draw ($(GString)!0.5!(1Aflat)$) + (0cm, 0.75cm) coordinate (0Top);
  \draw ($(EString)!0.5!(1Fflat)$) + (0cm,-0.75cm) coordinate (0Bottom);
  \draw[thick] (0Top) to (0Bottom);






\end{tikzpicture}



\end{document}
