\documentclass[crop,tikz]{standalone}% 'crop' is the default for v1.0, before it was 'preview'



% musical symbols
% \fl -> flat
% \sh -> sharp
\usepackage{musicography}

% tikz
\usepackage{tikz}
\usetikzlibrary{
    positioning
  , shapes
  , calc
}

\usepackage{lmodern}
\usepackage[T1]{fontenc}


\begin{document}
\tikzset{
  notePos/.style={
      draw
    , circle
    , thick
    , minimum size=1.5cm
  }
}

% A _ B ! C _ D _ E ! F _ G _ A

\begin{tikzpicture}[]

  % strings
  \node[notePos] (GString) {G};
  \node[notePos, below=of GString] (DString) {D};
  \node[notePos, below=of DString] (AString) {A};
  \node[notePos, below=of AString] (EString) {E};

  %  1st fret
  \node[notePos, right=of GString] (1A_) {A\fl / G\sh};
  \node[notePos, right=of DString] (1E_) {E\fl / D\sh};
  \node[notePos, right=of AString] (1B_) {B\fl / A\sh};
  \node[notePos, right=of EString] (1F) {F};

  % fret board beginning
  \draw ($(GString)!0.5!(1A_)$) + (0cm, 0.75cm) coordinate (0Top);
  \draw ($(EString)!0.5!(1F) $) + (0cm,-0.75cm) coordinate (0Bottom);
  \draw[thick] (0Top) to (0Bottom);

  %  2nd fret
  \node[notePos, right=of 1A_] (2A)  {A};
  \node[notePos, right=of 1E_] (2E)  {E};
  \node[notePos, right=of 1B_] (2B)  {B};
  \node[notePos, right=of 1F ] (2G_) {G\fl / F\sh};

  %  3rd fret
  \node[notePos, right=of 2A ] (3B_) {B\fl / A\sh};
  \node[notePos, right=of 2E ] (3F)  {F};
  \node[notePos, right=of 2B ] (3C)  {C};
  \node[notePos, right=of 2G_] (3G)  {G};

  \node[circle, fill, minimum size=0.75cm, below=0.5cm of 3G] (Marker3) {};

  %  4th fret
  \node[notePos, right=of 3B_] (4B)  {B};
  \node[notePos, right=of 3F ] (4G_) {G\fl / F\sh};
  \node[notePos, right=of 3C ] (4D_) {D\fl / C\sh};
  \node[notePos, right=of 3G ] (4A_) {A\fl / G\sh};

  %  5th fret
  \node[notePos, right=of 4B ] (5C) {C};
  \node[notePos, right=of 4G_] (5G)  {G};
  \node[notePos, right=of 4D_] (5D)  {D};
  \node[notePos, right=of 4A_] (5A)  {A};

  \node[circle, fill, minimum size=0.75cm, below=0.5cm of 5A] (Marker5) {};
\end{tikzpicture}



\end{document}
